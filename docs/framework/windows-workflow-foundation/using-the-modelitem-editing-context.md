## <a name="subscribing-to-a-service"></a><span data-ttu-id="8cd06-119">서비스 구독</span><span class="sxs-lookup"><span data-stu-id="8cd06-119">Subscribing to a service</span></span>

```csharp  
    if (!subscribed)  
    {  
        this.Context.Services.Subscribe<IMyService>(  
            servInstance =>  
            {  
                listBox1.ItemsSource = servInstance.GetValues(this.ModelItem.Properties["DisplayName"].ComputedValue.ToString());  
            }  
            );  
        subscribed = true;
    }  
}  
```
